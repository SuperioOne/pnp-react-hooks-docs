## Examples

```typescript
// get default view
const defaultView = useView("My List Title");

// get view by title

// get view by Id
const myView = useView("My List Title", "9db07c1f-7880-4601-99d0-1c39c43f6599");

// with query
const defaultViewInfo = useView("My List", undefined, {
	query: {
		select: ["Id"]
	}
```
